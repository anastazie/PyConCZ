def make_byte(four_letters):
    letters = {
        "A": 0b00, 
        "G": 0b01, 
        "C": 0b10, 
        "T": 0b11
    }
    return sum(
        [letters[letter] << (2 * i) 
            for i, letter 
            in enumerate(four_letters)])
